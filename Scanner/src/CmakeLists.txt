cmake_minimum_required(VERSION 3.10)

project(MalwareScannerProject LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

add_library(MalwareScanner SHARED
    MalwareScanner.cpp
)

add_executable(ScannerCLI ScannerCLI.cpp)

target_link_libraries(ScannerCLI PRIVATE MalwareScanner)
target_include_directories(ScannerCLI PRIVATE ${CMAKE_CURRENT_SOURCE_DIR})
target_include_directories(MalwareScanner PRIVATE ${CMAKE_CURRENT_SOURCE_DIR})

if (WIN32)
    target_compile_definitions(MalwareScanner PRIVATE MALWARESCANNER_EXPORTS)
endif()

if(MSVC)
    target_compile_options(MalwareScanner PRIVATE /W4 /WX)
    target_compile_options(ScannerCLI PRIVATE /W4 /WX)
else()
    target_compile_options(MalwareScanner PRIVATE -Wall -Wextra -Werror)
    target_compile_options(ScannerCLI PRIVATE -Wall -Wextra -Werror)
endif()
